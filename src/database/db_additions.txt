const getAllUsers = () => {
  return db.prepare('SELECT * FROM users ORDER BY rating DESC').all();
};

const banUser = (userId) => {
  return db.prepare('UPDATE users SET is_banned = 1 WHERE id = ?').run(userId);
};

const unbanUser = (userId) => {
  return db.prepare('UPDATE users SET is_banned = 0 WHERE id = ?').run(userId);
};

const isUserBanned = (userId) => {
  const user = getUser(userId);
  return user ? user.is_banned === 1 : false;
};

const addCosmetic = (userId, type, name, rarity) => {
  return db.prepare(
    INSERT INTO cosmetics (user_id, type, name, rarity) 
    VALUES (?, ?, ?, ?)
  ).run(userId, type, name, rarity);
};

const getBattlepass = (userId) => {
  return db.prepare('SELECT * FROM battlepass WHERE user_id = ?').get(userId);
};

const createBattlepass = (userId) => {
  return db.prepare(
    INSERT INTO battlepass (user_id, level, xp, is_premium) 
    VALUES (?, 0, 0, 0)
  ).run(userId);
};

const updateBattlepass = (userId, data) => {
  const fields = Object.keys(data).map(k => k + ' = ?').join(', ');
  const values = [...Object.values(data), userId];
  return db.prepare('UPDATE battlepass SET ' + fields + ' WHERE user_id = ?').run(...values);
};

const getActiveTournament = () => {
  return db.prepare('SELECT * FROM tournaments WHERE status = "active" OR status = "registration" LIMIT 1').get();
};

const addTournamentParticipant = (tournamentId, userId) => {
  return db.prepare('INSERT INTO tournament_participants (tournament_id, user_id) VALUES (?, ?)')
    .run(tournamentId, userId);
};
